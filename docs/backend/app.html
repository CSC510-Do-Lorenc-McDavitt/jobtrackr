@app.route("/download_file", methods=["POST"])
def download_file():

    '''
    ```
    View your files directly from cloud
    ```
    '''

    @after_this_request
    def delete(response):
        try:
            os.remove(request.get_json()["filename"].split("--;--")[1])
        except Exception:
            pass
        return response
    return files.download_file(Files)
